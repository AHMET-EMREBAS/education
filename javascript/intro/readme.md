### Functions

```javascript
/**
 * Set "abdefg" to the name property of the object.
 * @param {Object} obj0
 * @return {void}
 */
function mutateFunction(obj0) {
  obj0.name = "abdefg";
}

/**
 * Return the same object.
 * @param {Object} obj0
 * @returns {Object}
 */
function regularFunction(obj0) {
  return obj0;
}
```
